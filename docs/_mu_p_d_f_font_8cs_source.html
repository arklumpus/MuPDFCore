60 internal unsafe MuPDFFont(MuPDFContext context, IntPtr nativePointer)
61 {
62 int nameLength = 0;
63 int bold = -1;
64 int italic = -1;
65 int serif = -1;
66 int monospaced = -1;
67
68 ExitCodes result = (ExitCodes)NativeMethods.GetFontMetadata(context.NativeContext, nativePointer, ref nameLength, ref bold, ref italic, ref serif, ref monospaced);
69
70 switch (result)
71 {
72 case ExitCodes.EXIT_SUCCESS:
73 break;
74 case ExitCodes.ERR_FONT_METADATA:
75 throw new MuPDFException("An error occurred while retrieving font metadata!", result);
76 default:
77 throw new MuPDFException("Unknown error!", result);
78 }
79
80 byte[] fontName = new byte[nameLength];
81
82 fixed (byte* fontNamePtr = fontName)
83 {
84 result = (ExitCodes)NativeMethods.GetFontName(context.NativeContext, nativePointer, nameLength, (IntPtr)fontNamePtr);
85 }
86
87 switch (result)
88 {
89 case ExitCodes.EXIT_SUCCESS:
90 break;
91 case ExitCodes.ERR_FONT_METADATA:
92 throw new MuPDFException("An error occurred while retrieving font metadata!", result);
93 default:
94 throw new MuPDFException("Unknown error!", result);
95 }
96
97 this.IsBold = bold != 0;
98 this.IsItalic = italic != 0;
99 this.IsSerif = serif != 0;
100 this.IsMonospaced = monospaced != 0;
101 this.Name = Encoding.UTF8.GetString(fontName);
102
103 this.OwnerContext = context;
104 this.NativePointer = nativePointer;
105 }

107 /// <summary>
108 /// Get a pointer to the FreeType FT_Face object for this font. You will need native bindings to the FreeType library to use this.
109 /// </summary>
110 /// <returns>A pointer to the FreeType FT_Face object for this font, or <see cref="IntPtr.Zero"/> if this is a Type3 font.</returns>
111 /// <exception cref="MuPDFException">Thrown if an error occurs while accessing the FreeType handle for the font.</exception>
112 public IntPtr GetFreeTypeHandle()
113 {
114 if (disposedValue)
115 {
116 throw new ObjectDisposedException("MuPDFFont", "The MuPDFFont object has already been disposed! Maybe you disposed the MuPDFStructuredTextPage that contained it?");
117 }
118
119 IntPtr tbr = IntPtr.Zero;
120
121 ExitCodes result = (ExitCodes)NativeMethods.GetFTHandle(this.OwnerContext.NativeContext, this.NativePointer, ref tbr);
122
123 switch (result)
124 {
125 case ExitCodes.EXIT_SUCCESS:
126 break;
127 case ExitCodes.ERR_FONT_METADATA:
128 throw new MuPDFException("Cannot get the font handle!", result);
129 default:
130 throw new MuPDFException("Unknown error", result);
131 }
132
133 return tbr;
134 }

165 /// <inheritdoc/>
166 protected virtual void Dispose(bool disposing)
167 {
168 if (!disposedValue)
169 {
170 if (OwnerContext.disposedValue)
171 {
172 throw new LifetimeManagementException<MuPDFFont, MuPDFContext>(this, OwnerContext, this.NativePointer, OwnerContext.NativeContext);
173 }
174
175 lock (OwnerContext.FontCacheLock)
176 {
177 (MuPDFFont font, int referenceCount) item;
178 if (OwnerContext.FontCache.TryGetValue(this.NativePointer, out item))
179 {
180 if (item.referenceCount <= 1)
181 {
182 NativeMethods.DisposeFont(this.OwnerContext.NativeContext, this.NativePointer);
183 OwnerContext.FontCache.Remove(this.NativePointer);
184 disposedValue = true;
185 }
186 else
187 {
188 item = (item.font, item.referenceCount - 1);
189 OwnerContext.FontCache[this.NativePointer] = item;
190 }
191 }
192 }
193 }
