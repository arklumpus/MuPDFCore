18 using System;

20 namespace MuPDFCore
21 {
22  /// <summary>
23  /// A wrapper around a MuPDF display list object, which contains the necessary informations to render a page to an image.
24  /// </summary>
25  internal class MuPDFDisplayList : IDisposable
26  {
27  /// <summary>
28  /// The context that owns the document that was used to create this display list.
29  /// </summary>
30  private readonly MuPDFContext OwnerContext;
31 
32  /// <summary>
33  /// A pointer to the native display list object.
34  /// </summary>
35  readonly internal IntPtr NativeDisplayList;
36 
37  /// <summary>
38  /// The display list's bounds in page units. Read-only.
39  /// </summary>
40  public Rectangle Bounds { get; }
41 
42  /// <summary>
43  /// Create a new <see cref="MuPDFDisplayList"/> instance from the specified page.
44  /// </summary>
45  /// <param name="context">The context that owns the document from which the page was taken.</param>
46  /// <param name="page">The page from which the display list should be generated.</param>
47  /// <param name="includeAnnotations">If this is <see langword="true" />, annotations (e.g. signatures) are included in the display list that is generated. Otherwise, only the page contents are included.</param>
48  public MuPDFDisplayList(MuPDFContext context, MuPDFPage page, bool includeAnnotations = true)
49  {
50  this.OwnerContext = context;
51 
52  float x0 = 0;
53  float y0 = 0;
54  float x1 = 0;
55  float y1 = 0;
56 
57  ExitCodes result = (ExitCodes)NativeMethods.GetDisplayList(context.NativeContext, page.NativePage, includeAnnotations ? 1 : 0, ref NativeDisplayList, ref x0, ref y0, ref x1, ref y1);
58 
59  switch (result)
60  {
61  case ExitCodes.EXIT_SUCCESS:
62  break;
63  case ExitCodes.ERR_CANNOT_RENDER:
64  throw new MuPDFException("Cannot render page", result);
65  default:
66  throw new MuPDFException("Unknown error", result);
67  }
68 
69  this.Bounds = new Rectangle(Math.Round(x0 * page.OwnerDocument.ImageXRes / 72.0 * 1000) / 1000, Math.Round(y0 * page.OwnerDocument.ImageYRes / 72.0 * 1000) / 1000, Math.Round(x1 * page.OwnerDocument.ImageXRes / 72.0 * 1000) / 1000, Math.Round(y1 * page.OwnerDocument.ImageYRes / 72.0 * 1000) / 1000);
70  }
71 
72  private bool disposedValue;
73 
74  protected virtual void Dispose(bool disposing)
75  {
76  if (!disposedValue)
77  {
78  NativeMethods.DisposeDisplayList(OwnerContext.NativeContext, NativeDisplayList);
79  disposedValue = true;
80  }
81  }
82 
83  ~MuPDFDisplayList()
84  {
85  Dispose(disposing: false);
86  }
87 
88  public void Dispose()
89  {
90  Dispose(disposing: true);
91  GC.SuppressFinalize(this);
92  }
93  }
94 }
