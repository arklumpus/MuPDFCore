85 /// <summary>

147 /// <summary>
148 /// Create a new <see cref="MuPDFContext"/> instance with the specified cache store size.
149 /// </summary>
150 /// <param name="storeSize">The maximum size in bytes of the resource cache store. The default value is 256 MiB.</param>
151 public MuPDFContext(uint storeSize = 256 << 20)
152 {
153 ExitCodes result = (ExitCodes)NativeMethods.CreateContext((ulong)storeSize, ref NativeContext);
154
155 switch (result)
156 {
157 case ExitCodes.EXIT_SUCCESS:
158 break;
159 case ExitCodes.ERR_CANNOT_CREATE_CONTEXT:
160 throw new MuPDFException("Cannot create MuPDF context", result);
161 case ExitCodes.ERR_CANNOT_REGISTER_HANDLERS:
162 throw new MuPDFException("Cannot register document handlers", result);
163 default:
164 throw new MuPDFException("Unknown error", result);
165 }
166 }

168 /// <summary>
169 /// Wrap an existing pointer to a native MuPDF context object.
170 /// </summary>

172 /// <param name="nativeContext">The pointer to the native context that should be used.</param>
173 internal MuPDFContext(MuPDFContext parentContext, IntPtr nativeContext)
174 {
175 this.NativeContext = nativeContext;

179 /// <summary>
180 /// Evict all items from the resource cache store (freeing the memory where they were held).
181 /// </summary>
182 public void ClearStore()
183 {
184 NativeMethods.EmptyStore(this.NativeContext);
185 }
186
187 /// <summary>
188 /// Evict items from the resource cache store (freeing the memory where they were held) until the the size of the store drops to the specified fraction of the current size.
189 /// </summary>
190 /// <param name="fraction">The fraction of the current size that constitutes the target size of the store. If this is &lt;= 0, the cache is cleared. If this is &gt;= 1, nothing happens.</param>
191 public void ShrinkStore(double fraction)
192 {
193 if (fraction <= 0)
194 {
195 ClearStore();
196 }
197 else if (Math.Round(fraction * 100) < 100)
198 {
199 NativeMethods.ShrinkStore(this.NativeContext, (uint)Math.Round(fraction * 100));
200 }
201 }
202
203 /// <summary>
204 /// Resolve a font from the font cache, or create a new font object.
205 /// </summary>
206 /// <param name="nativePointer">The pointer to the font.</param>
207 /// <returns>The cached font instance, or a new font object.</returns>
208 internal MuPDFFont Resolve(IntPtr nativePointer)
209 {
210 lock (FontCacheLock)
211 {
212 (MuPDFFont font, int referenceCount) item;
213
214 if (!this.FontCache.TryGetValue(nativePointer, out item) || item.font.disposedValue)
215 {
216 item = (new MuPDFFont(this, nativePointer), 1);
217 this.FontCache[nativePointer] = item;
218 }
219 else
220 {
221 item = (item.font, item.referenceCount + 1);
222 this.FontCache[nativePointer] = item;
223 }
224
225 return item.font;
226 }
227 }

229 internal bool disposedValue;
230
231 ///<inheritdoc/>
232 protected virtual void Dispose(bool disposing)
233 {
234 if (!disposedValue)
235 {
236 if (disposing && this.ParentContext == null)
237 {
238 lock (fontCacheLock)
239 {
240 this.fontCache.Clear();
241 }
242
243 this.fontCache = null;
244 }
245
246 NativeMethods.DisposeContext(NativeContext);
247 disposedValue = true;
248 }
249 }

251 ///<inheritdoc/>
252 ~MuPDFContext()
253 {
254 Dispose(disposing: false);
255 }
256
257 ///<inheritdoc/>
258 public void Dispose()
259 {
260 Dispose(disposing: true);
261 GC.SuppressFinalize(this);
262 }
