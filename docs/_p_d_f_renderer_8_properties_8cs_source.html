125 /// <summary>
126 /// Defines the <see cref="DisplayArea"/> property.
127 /// </summary>
128 public static readonly StyledProperty<Rect> DisplayAreaProperty = AvaloniaProperty.Register<PDFRenderer, Rect>(nameof(DisplayArea));
129 /// <summary>
130 /// The region of the page (in page units) that is currently displayed by the current instance. This always has the same aspect ratio of the bounds of this control.
131 /// When this is set, the value is sanitised so that the smallest rectangle with the correct aspect ratio containing the requested value is chosen.
132 /// </summary>
133 public Rect DisplayArea
134 {
135 get
136 {
137 return GetValue(DisplayAreaProperty);
138 }
139
140 set
141 {
142 double widthRatio = value.Width / (this.Bounds.Width);
143 double heightRatio = value.Height / (this.Bounds.Height);
144
145 double containingWidth = Math.Max(widthRatio, heightRatio) * this.Bounds.Width;
146 double containingHeight = Math.Max(widthRatio, heightRatio) * this.Bounds.Height;
147
148 double deltaW = (containingWidth - value.Width) * 0.5;
149 double deltaH = (containingHeight - value.Height) * 0.5;
150
151 Rect newDispArea = new Rect(new Point(value.X - deltaW, value.Y - deltaH), new Point(value.Right + deltaW, value.Bottom + deltaH));
152
153 SetValue(DisplayAreaProperty, newDispArea);
154 }
155 }

205 /// <summary>
206 /// Defines the <see cref="Zoom"/> property.
207 /// </summary>
208 public static readonly DirectProperty<PDFRenderer, double> ZoomProperty = AvaloniaProperty.RegisterDirect<PDFRenderer, double>(nameof(Zoom), o => o.Zoom, (o, v) => o.Zoom = v, defaultBindingMode: Avalonia.Data.BindingMode.TwoWay);
209 /// <summary>
210 /// Backing field for the <see cref="Zoom"/> property.
211 /// </summary>
212 private double _Zoom;
213 /// <summary>
214 /// The current zoom level. Setting this will change the <see cref="DisplayArea"/> appropriately, zooming around the center of the <see cref="DisplayArea"/>.
215 /// </summary>
216 public double Zoom
217 {
218 get
219 {
220 return _Zoom;
221 }
222
223 set
224 {
225 double actualZoom = value / _Zoom;
226
227 double currZoomX = FixedArea.Width / DisplayArea.Width * actualZoom;
228 double currZoomY = FixedArea.Height / DisplayArea.Height * actualZoom;
229
230 double currWidth = FixedArea.Width / currZoomX;
231 double currHeight = FixedArea.Height / currZoomY;
232
233
234 Point pos = new Point(this.Bounds.Width * 0.5, this.Bounds.Height * 0.5);
235
236 double deltaW = currWidth - DisplayArea.Width;
237 double deltaH = currHeight - DisplayArea.Height;
238
239 SetValue(DisplayAreaProperty, new Rect(new Point(DisplayArea.X - deltaW * pos.X / this.Bounds.Width, DisplayArea.Y - deltaH * pos.Y / this.Bounds.Height), new Point(DisplayArea.Right + deltaW * (1 - pos.X / this.Bounds.Width), DisplayArea.Bottom + deltaH * (1 - pos.Y / this.Bounds.Height))));
240 }
