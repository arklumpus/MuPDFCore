39 /// <summary>
40 /// Load the outline for a document.
41 /// </summary>
42 /// <param name="context">A <see cref="MuPDFContext"/> to store resources and the exception stack.</param>
43 /// <param name="document">The document whose outline should be loaded.</param>
44 internal MuPDFOutline(MuPDFContext context, MuPDFDocument document)
45 {
46 this.OwnerDocument = document;
47 IntPtr outline = NativeMethods.LoadOutline(context.NativeContext, document.NativeDocument);
48
49 if (outline != IntPtr.Zero)
50 {
51 this.Items = MuPDFOutlineItem.TraverseOutline(outline, this).ToArray();
52 NativeMethods.DisposeOutline(context.NativeContext, outline);
53 }
54 else
55 {
56 this.Items = new MuPDFOutlineItem[0];
57 }

150 /// <summary>
151 /// Get the length of a null-terminated C string.
152 /// </summary>
153 /// <param name="stringAddress">A pointer to the string.</param>
154 /// <returns>The length of the string in bytes.</returns>
155 private static unsafe int strlen(byte* stringAddress)
156 {
157 byte* originalAddress = stringAddress;
158
159 while (*stringAddress != 0)
160 {
161 stringAddress++;
162 }
163
164 return (int)(stringAddress - originalAddress);
165 }
166
167 /// <summary>
168 /// Convert a null-terminated C string in UTF8 encoding to a .NET <see langword="string"/>.
169 /// </summary>
170 /// <param name="stringAddress">A pointer to the string.</param>
171 /// <returns>A .NET <see langword="string"/>.</returns>
172 private static unsafe string PtrToStringUTF8(byte* stringAddress) => Encoding.UTF8.GetString(stringAddress, strlen(stringAddress));
173
174 /// <summary>
175 /// Create a new <see cref="MuPDFOutlineItem"/> from a native pointer.
176 /// </summary>
177 /// <param name="nativeOutlineItem">The pointer to the native outline item.</param>
178 /// <param name="next">A pointer to the next item in the outline. This will be <see cref="IntPtr.Zero"/> for the last item.</param>
179 /// <param name="ownerOutline">The <see cref="MuPDFOutline"/> that contains the current item.</param>
180 private unsafe MuPDFOutlineItem(IntPtr nativeOutlineItem, out IntPtr next, MuPDFOutline ownerOutline)
181 {
182 fz_outline nativeItem = Marshal.PtrToStructure<fz_outline>(nativeOutlineItem);
183
184 string title = nativeItem.title == IntPtr.Zero ? null : PtrToStringUTF8((byte*)nativeItem.title);
185 string uri = nativeItem.uri == IntPtr.Zero ? null : PtrToStringUTF8((byte*)nativeItem.uri);
186
187 this.Title = title;
188 this.Uri = uri;
189 this.Chapter = nativeItem.chapter;
190 this.Page = nativeItem.page;
191 this.Location = new PointF(nativeItem.x, nativeItem.y);
192 this.OwnerOutline = ownerOutline;
193
194 next = nativeItem.next;
195
196 if (nativeItem.down != IntPtr.Zero)
197 {
198 this.Children = TraverseOutline(nativeItem.down, ownerOutline).ToArray();
199 }
200 else
201 {
202 this.Children = new MuPDFOutlineItem[0];
203 }
204 }
205
206 /// <summary>
207 /// Recursively traverse the outline, loading information about all the items.
208 /// </summary>
209 /// <param name="firstOutlineItem">The first item in the outline.</param>
210 /// <param name="ownerOutline">The <see cref="MuPDFOutline"/> that contains the current item.</param>
211 /// <returns>A collection of <see cref="MuPDFOutlineItem"/> containing all the information in the outline.</returns>
212 internal static IEnumerable<MuPDFOutlineItem> TraverseOutline(IntPtr firstOutlineItem, MuPDFOutline ownerOutline)
213 {
214 IntPtr currOutlineItem = firstOutlineItem;
215
216 while (currOutlineItem != IntPtr.Zero)
217 {
218 MuPDFOutlineItem item = new MuPDFOutlineItem(currOutlineItem, out IntPtr next, ownerOutline);
219 yield return item;
220 currOutlineItem = next;
221 }
222 }
223 }
