25 /// <summary>
26 /// A collection of <see cref="MuPDFLink"/> objects.
27 /// </summary>
28 public class MuPDFLinks : IReadOnlyList<MuPDFLink>, IDisposable
29 {
30 private bool disposedValue;
31
32 /// <inheritdoc/>
33 public MuPDFLink this[int index] => Contents[index];
34
35 /// <inheritdoc/>
36 public int Count => Contents.Length;
37
38 internal MuPDFPage OwnerPage { get; }
39 private IntPtr FirstLinkPointer { get; }
40
41 internal MuPDFLink[] Contents { get; }
42
43 internal unsafe MuPDFLinks(MuPDFPage ownerPage)
44 {
45 this.OwnerPage = ownerPage;
46
47 IntPtr firstLink = IntPtr.Zero;
48 int countLinks = NativeMethods.CountLinks(ownerPage.OwnerDocument.OwnerContext.NativeContext, ownerPage.NativePage, ref firstLink);
49
50 if (countLinks > 0)
51 {
52 this.FirstLinkPointer = firstLink;
53 int[] uriLengths = new int[countLinks];
54 IntPtr[] linkPointers = new IntPtr[countLinks];
55
56 fixed (int* uriLengthsPtr = uriLengths)
57 fixed (IntPtr* linkPointersPtr = linkPointers)
58 {
59 NativeMethods.LoadLinks(firstLink, (IntPtr)linkPointersPtr, (IntPtr)uriLengthsPtr);
60 }
61
62 this.Contents = new MuPDFLink[countLinks];
63
64 for (int i = 0; i < countLinks; i++)
65 {
66 this.Contents[i] = new MuPDFLink(this, linkPointers[i], uriLengths[i]);
67 }
68 }
69 else
70 {
71 this.Contents = new MuPDFLink[0];
72 this.FirstLinkPointer = IntPtr.Zero;
73 }
74 }
75
76 /// <inheritdoc/>
77 public IEnumerator<MuPDFLink> GetEnumerator()
78 {
79 return ((IEnumerable<MuPDFLink>)Contents).GetEnumerator();
80 }
81
82 IEnumerator IEnumerable.GetEnumerator()
83 {
84 return Contents.GetEnumerator();
85 }
86
87 /// <inheritdoc/>
88 protected virtual void Dispose(bool disposing)
89 {
90 if (!disposedValue)
91 {
92 if (this.FirstLinkPointer != IntPtr.Zero)
93 {
94 NativeMethods.DisposeLinks(this.OwnerPage.OwnerDocument.OwnerContext.NativeContext, this.FirstLinkPointer);
95 }
96 disposedValue = true;
97 }
98 }
99
100 /// <inheritdoc/>
101 ~MuPDFLinks()
102 {
103 Dispose(disposing: false);
104 }
105
106 /// <summary>
107 /// <inheritdoc/>
108 /// </summary>
109 public void Dispose()
110 {
111 Dispose(disposing: true);
112 GC.SuppressFinalize(this);
113 }

116 /// <summary>
117 /// Represents a link within a MuPDF document.
118 /// </summary>
119 public class MuPDFLink
120 {
121 /// <summary>
122 /// The area on which users can click to activate the link.
123 /// </summary>
124 public Rectangle ActiveArea { get; }
125
126 /// <summary>
127 /// The link destination.
128 /// </summary>
129 public MuPDFLinkDestination Destination { get; }
130
131 /// <summary>
132 /// Whether the link is visible or hidden (e.g., because it is part of a hidden optional content group).
133 /// </summary>
134 public bool IsVisible => this.OwnerCollection.OwnerPage.OwnerDocument.NativePDFDocument == IntPtr.Zero || NativeMethods.IsLinkHidden(this.OwnerCollection.OwnerPage.OwnerDocument.OwnerContext.NativeContext, "View", this.NativeLink) == 0;
135
136 private string Uri { get; }
137 internal IntPtr NativeLink { get; }
138 internal MuPDFLinks OwnerCollection { get; }
139
140 internal unsafe MuPDFLink(MuPDFLinks ownerCollection, IntPtr linkPointer, int uriLength)
141 {
142 this.OwnerCollection = ownerCollection;
143 float x0 = 0;
144 float x1 = 0;
145 float y0 = 0;
146 float y1 = 0;
147
148 byte[] uriBytes = new byte[uriLength];
149
150 int isExternal = -1;
151 int isSetOCGState = -1;
152
153 int destinationType = -1;
154 float x = 0;
155 float y = 0;
156 float w = 0;
157 float h = 0;
158 float zoom = 0;
159 int page = -1;
160 int chapter = -1;
161
162 fixed (byte* uriPtr = uriBytes)
163 {
164 NativeMethods.LoadLink(ownerCollection.OwnerPage.OwnerDocument.OwnerContext.NativeContext, ownerCollection.OwnerPage.OwnerDocument.NativeDocument, linkPointer, uriLength, ownerCollection.OwnerPage.OwnerDocument.NativePDFDocument != IntPtr.Zero ? 1 : 0, ref x0, ref y0, ref x1, ref y1, (IntPtr)uriPtr, ref isExternal, ref isSetOCGState, ref destinationType, ref x, ref y, ref w, ref h, ref zoom, ref chapter, ref page);
165 }
166
167 this.ActiveArea = new Rectangle(x0, y0, x1, y1);
168
169 string uri = Encoding.UTF8.GetString(uriBytes);
170
171 this.Uri = uri;
172 this.NativeLink = linkPointer;
173
174 if (isExternal != 0)
175 {
176 this.Destination = new MuPDFExternalLinkDestination(this, uri);
177 }
178 else if (isSetOCGState != 0)
179 {
180 this.Destination = new MuPDFSetOCGStateLinkDestination(this);
181 }
182 else
183 {
184 this.Destination = new MuPDFInternalLinkDestination(this, page, chapter, x, y, w, h, zoom, destinationType);
185 }
186 }
187 }
