294  /// <summary>
295  /// An <see cref="IDisposable"/> wrapper around an <see cref="IntPtr"/> that frees the allocated memory when it is disposed.
296  /// </summary>
297  public class DisposableIntPtr : IDisposable
298  {
299  /// <summary>
300  /// The pointer to the unmanaged memory.
301  /// </summary>
302  private readonly IntPtr InternalPointer;
303 
304  /// <summary>
305  /// Create a new DisposableIntPtr.
306  /// </summary>
307  /// <param name="pointer">The pointer that should be freed upon disposing of this object.</param>
308  public DisposableIntPtr(IntPtr pointer)
309  {
310  this.InternalPointer = pointer;
311  }
312 
313  private bool disposedValue;
314 
315  ///<inheritdoc/>
316  protected virtual void Dispose(bool disposing)
317  {
318  if (!disposedValue)
319  {
320  Marshal.FreeHGlobal(InternalPointer);
321  disposedValue = true;
322  }
323  }
324 
325  ///<inheritdoc/>
326  ~DisposableIntPtr()
327  {
328  Dispose(disposing: false);
329  }
330 
331  ///<inheritdoc/>
332  public void Dispose()
333  {
334  Dispose(disposing: true);
335  GC.SuppressFinalize(this);
336  }
337  }
