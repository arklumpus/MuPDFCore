26 /// <summary>
27 /// A reusable thread to render part of an image.
28 /// </summary>
29 internal class RenderingThread : IDisposable

31 /// <summary>
32 /// A class to hold the data used internally by the rendering thread.
33 /// </summary>
34 private class RenderData

36 public IntPtr Context;
37 public MuPDFDisplayList DisplayList;
38 public Rectangle Region;

47 /// A lock object to prevent race conditions.
48 /// </summary>
49 private readonly object RenderDataLock;

52 /// The data used internally by the rendering thread.
53 /// </summary>

57 /// The actual thread that does the rendering.
58 /// </summary>

62 /// An <see cref="EventWaitHandle"/> that is set by the <see cref="Thread"/> when it finished rendering.
63 /// </summary>

66 /// <summary>
67 /// An <see cref="EventWaitHandle"/> that signals to the <see cref="Thread"/> that it should start rendering.
68 /// </summary>

72 /// An <see cref="EventWaitHandle"/> that signals that the object is being disposed and all activity should cease.
73 /// </summary>
74 private readonly EventWaitHandle DisposeSignal;

77 /// A pointer to a <see cref="Cookie"/> object that can be used to monitor the progress of the rendering or to abort it.

81 /// <summary>
82 /// Performs the actual rendering.
83 /// </summary>
84 private void RenderAction()
85 {
86 ExitCodes result = (ExitCodes)NativeMethods.RenderSubDisplayList(this.CurrentRenderData.Context, this.CurrentRenderData.DisplayList.NativeDisplayList, this.CurrentRenderData.Region.X0, this.CurrentRenderData.Region.Y0, this.CurrentRenderData.Region.X1, this.CurrentRenderData.Region.Y1, this.CurrentRenderData.Zoom, (int)this.CurrentRenderData.PixelFormat, this.CurrentRenderData.PixelStorage, Cookie);
87
88 switch (result)
89 {
90 case ExitCodes.EXIT_SUCCESS:
91 break;
92 case ExitCodes.ERR_CANNOT_RENDER:
93 throw new MuPDFException("Cannot render page", result);
94 default:
95 throw new MuPDFException("Unknown error", result);
96 }
97
98 RoundedRectangle roundedRegion = this.CurrentRenderData.Region.Round(this.CurrentRenderData.Zoom);
99 RoundedSize roundedSize = new RoundedSize(roundedRegion.Width, roundedRegion.Height);
100
101 if (this.CurrentRenderData.PixelFormat == PixelFormats.RGBA || this.CurrentRenderData.PixelFormat == PixelFormats.BGRA)
102 {
103 Utils.UnpremultiplyAlpha(this.CurrentRenderData.PixelStorage, roundedSize);
104 }
105
106 if (this.CurrentRenderData.ClipToPageBounds && !this.CurrentRenderData.PageBounds.Contains(this.CurrentRenderData.DisplayList.Bounds.Intersect(this.CurrentRenderData.Region)))
107 {
108 Utils.ClipImage(this.CurrentRenderData.PixelStorage, roundedSize, this.CurrentRenderData.Region, this.CurrentRenderData.PageBounds, this.CurrentRenderData.PixelFormat);
109 }
110 }

112 /// <summary>
113 /// Create a new <see cref="RenderingThread"/> instance.
114 /// </summary>
115 public RenderingThread()
116 {
117 //Initialize fields
118 SignalFromThread = new EventWaitHandle(false, EventResetMode.ManualReset);
119 SignalToThread = new EventWaitHandle(false, EventResetMode.ManualReset);
120 DisposeSignal = new EventWaitHandle(false, EventResetMode.ManualReset);
121
122 CurrentRenderData = new RenderData();
123 RenderDataLock = new object();
124
125 //Allocate unmanaged memory to hold the cookie.
126 Cookie = Marshal.AllocHGlobal(Marshal.SizeOf<MuPDFCore.Cookie>());
127
128 //Initialise the rendering thread.
129 this.Thread = new Thread(() =>
130 {
131 EventWaitHandle[] handles = new EventWaitHandle[] { SignalToThread, DisposeSignal };
132
133 while (true)
134 {
135 int handle = EventWaitHandle.WaitAny(handles);
136
137 if (handle == 0)
138 {
139 SignalToThread.Reset();
140
141 lock (RenderDataLock)
142 {
143 this.RenderAction();
144 }
145
146 SignalFromThread.Set();
147 }
148 else
149 {
150 SignalFromThread.Set();
151 break;
152 }
153 }
154 });
155
156 //Start the rendering thread.
157 this.Thread.Start();
158 }

160 /// <summary>
161 /// Start rendering a region of a display list to the specified destination.
162 /// </summary>
163 /// <param name="context">The rendering context.</param>
164 /// <param name="displayList">The native display list that should be rendered.</param>
165 /// <param name="region">The region that should be rendered.</param>
166 /// <param name="zoom">The scale at which the region will be rendered. This will determine the size in pixel of the image.</param>
167 /// <param name="pixelStorage">The address of the buffer where the pixel data will be written. There must be enough space available to write the values for all the pixels, otherwise this will fail catastrophically!</param>
168 /// <param name="pixelFormat">The format of the pixel data.</param>
169 /// <param name="pageBounds">The bounds of the page being rendererd.</param>
170 /// <param name="clipToPageBounds">A boolean value indicating whether the rendered image should be clipped to the original page's bounds. This can be relevant if the page has been "cropped" by altering its mediabox, but otherwise leaving the contents untouched.</param>
171 public void Render(IntPtr context, MuPDFDisplayList displayList, Rectangle region, float zoom, IntPtr pixelStorage, PixelFormats pixelFormat, Rectangle pageBounds, bool clipToPageBounds)
172 {
173 lock (RenderDataLock)
174 {
175 //Reset the cookie.
176 unsafe
177 {
178 Cookie* cookie = (Cookie*)Cookie;
179
180 cookie->abort = 0;
181 cookie->errors = 0;
182 cookie->incomplete = 0;
183 cookie->progress = 0;
184 cookie->progress_max = 0;
185 }
186
187 //Set up all the rendering data.
188 CurrentRenderData.Context = context;
189 CurrentRenderData.DisplayList = displayList;
190 CurrentRenderData.Region = region;
191 CurrentRenderData.Zoom = zoom;
192 CurrentRenderData.PixelStorage = pixelStorage;
193 CurrentRenderData.PixelFormat = pixelFormat;
194 CurrentRenderData.PageBounds = pageBounds;
195 CurrentRenderData.ClipToPageBounds = clipToPageBounds;
196 SignalToThread.Set();
197 }
198 }

200 /// <summary>
201 /// Wait until the current rendering operation finishes.
202 /// </summary>
203 public void WaitForRendering()
204 {
205 EventWaitHandle[] handles = new EventWaitHandle[] { SignalFromThread, DisposeSignal };
206 int result = EventWaitHandle.WaitAny(handles);
207 if (result == 0)
208 {
209 SignalFromThread.Reset();
210 }
211 }

273 /// <summary>
274 /// A class that holds the necessary resources to render a page of a MuPDF document using multiple threads.
275 /// </summary>
276 public class MuPDFMultiThreadedPageRenderer : IDisposable
277 {
278 /// <summary>
279 /// The display list that is rendered by this renderer.
280 /// </summary>
281 private readonly MuPDFDisplayList DisplayList;
282
283 /// <summary>
284 /// The cloned contexts that are used by the <see cref="RenderingThreads"/> to render the display list.
285 /// </summary>
286 private readonly MuPDFContext[] Contexts;
287
288 /// <summary>
289 /// The <see cref="RenderingThreads"/> that are in charge of the actual rendering.
290 /// </summary>
291 private readonly RenderingThread[] RenderingThreads;
292
293 /// <summary>
294 /// The bounds of the page being rendered.
295 /// </summary>
296 private readonly Rectangle PageBounds;
297
298 /// <summary>
299 /// Whether the rendered images should be clipped to the bounds of the page being rendered.
300 /// </summary>
301 private readonly bool ClipToPageBounds;

303 /// <summary>
304 /// The number of threads that are used to render the image.
305 /// </summary>
306 public int ThreadCount { get; }

308 /// <summary>
309 /// If the document is an image, the horizontal resolution of the image. Otherwise, 72.
310 /// </summary>
311 internal double ImageXRes = double.NaN;
312
313 /// <summary>
314 /// If the document is an image, the vertical resolution of the image. Otherwise, 72.
315 /// </summary>
316 internal double ImageYRes = double.NaN;

318 /// <summary>
319 /// Create a new <see cref="MuPDFMultiThreadedPageRenderer"/> from a specified display list using the specified number of threads.
320 /// </summary>
321 /// <param name="context">The context that owns the document from which the display list was extracted.</param>
322 /// <param name="displayList">The display list to render.</param>
323 /// <param name="threadCount">The number of threads to use in the rendering. This must be factorisable using only powers of 2, 3, 5 or 7. Otherwise, the biggest number smaller than <paramref name="threadCount"/> that satisfies this condition is used.</param>
324 /// <param name="pageBounds">The bounds of the page being rendererd.</param>
325 /// <param name="clipToPageBounds">A boolean value indicating whether the rendered image should be clipped to the original page's bounds. This can be relevant if the page has been "cropped" by altering its mediabox, but otherwise leaving the contents untouched.</param>
326 /// <param name="imageXRes">If the document is an image, the horizontal resolution of the image. Otherwise, 72.</param>
327 /// <param name="imageYRes">If the document is an image, the vertical resolution of the image. Otherwise, 72.</param>
328 internal MuPDFMultiThreadedPageRenderer(MuPDFContext context, MuPDFDisplayList displayList, int threadCount, Rectangle pageBounds, bool clipToPageBounds, double imageXRes, double imageYRes)
329 {
330 threadCount = Utils.GetAcceptableNumber(threadCount);
331
332 this.ThreadCount = threadCount;
333 this.DisplayList = displayList;
334 this.PageBounds = pageBounds;
335 this.ClipToPageBounds = clipToPageBounds;
336
337 this.ImageXRes = imageXRes;
338 this.ImageYRes = imageYRes;
339
340 IntPtr[] contexts = new IntPtr[threadCount];
341 GCHandle contextsHandle = GCHandle.Alloc(contexts, GCHandleType.Pinned);
342
343 try
344 {
345 ExitCodes result = (ExitCodes)NativeMethods.CloneContext(context.NativeContext, threadCount, contextsHandle.AddrOfPinnedObject());
346
347 switch (result)
348 {
349 case ExitCodes.EXIT_SUCCESS:
350 break;
351 case ExitCodes.ERR_CANNOT_INIT_MUTEX:
352 throw new MuPDFException("Cannot initalize mutex objects", result);
353 case ExitCodes.ERR_CANNOT_CREATE_CONTEXT:
354 throw new MuPDFException("Cannot create master context", result);
355 case ExitCodes.ERR_CANNOT_CLONE_CONTEXT:
356 throw new MuPDFException("Cannot create context clones", result);
357 default:
358 throw new MuPDFException("Unknown error", result);
359 }
360
361 Contexts = new MuPDFContext[threadCount];
362 RenderingThreads = new RenderingThread[threadCount];
363 for (int i = 0; i < threadCount; i++)
364 {
365 Contexts[i] = new MuPDFContext(contexts[i]);
366 RenderingThreads[i] = new RenderingThread();
367 }
368 }
369 finally
370 {
371 contextsHandle.Free();
372 }
373 }

375 /// <summary>
376 /// Render the specified region to an image of the specified size, split in a number of tiles equal to the number of threads used by this <see cref="MuPDFMultiThreadedPageRenderer"/>, without marshaling. This method will not return until all the rendering threads have finished.
377 /// </summary>
378 /// <param name="targetSize">The total size of the image that should be rendered.</param>
379 /// <param name="region">The region in page units that should be rendered.</param>
380 /// <param name="destinations">An array containing the addresses of the buffers where the rendered tiles will be written. There must be enough space available in each buffer to write the values for all the pixels of the tile, otherwise this will fail catastrophically!
381 /// As long as the <paramref name="targetSize"/> is the same, the size in pixel of the tiles is guaranteed to also be the same.</param>
382 /// <param name="pixelFormat">The format of the pixel data.</param>
383 public void Render(RoundedSize targetSize, Rectangle region, IntPtr[] destinations, PixelFormats pixelFormat)
384 {
385 if (destinations.Length != Contexts.Length)
386 {
387 throw new ArgumentOutOfRangeException(nameof(destinations), destinations.Length, "The number of destinations must be equal to the number of rendering threads!");
388 }
389
390 RoundedRectangle[] targets = targetSize.Split(destinations.Length);
391
392 float zoomX = targetSize.Width / region.Width;
393 float zoomY = targetSize.Height / region.Height;
394
395 float zoom = (float)Math.Sqrt(zoomX * zoomY);
396
397 Rectangle actualPageArea = new Rectangle(region.X0, region.Y0, region.X0 + targetSize.Width / zoom, region.Y0 + targetSize.Height / zoom);
398
399 Rectangle[] origins = actualPageArea.Split(destinations.Length);
400
401 //Make sure that each tile has the expected size in pixel, rounding errors notwithstanding.
402 for (int i = 0; i < origins.Length; i++)
403 {
404 int countBlanks = 0;
405 RoundedRectangle roundedOrigin = origins[i].Round(zoom);
406 while (roundedOrigin.Width != targets[i].Width || roundedOrigin.Height != targets[i].Height)
407 {
408 RoundedRectangle oldRoundedOrigin = roundedOrigin;
409
410 if (roundedOrigin.Width > targets[i].Width)
411 {
412 if (origins[i].X0 > actualPageArea.X0)
413 {
414 origins[i] = new Rectangle(origins[i].X0 + 0.5 / zoom, origins[i].Y0, origins[i].X1, origins[i].Y1);
415 }
416 else
417 {
418 origins[i] = new Rectangle(origins[i].X0, origins[i].Y0, origins[i].X1 - 0.5 / zoom, origins[i].Y1);
419 }
420 }
421 else if (roundedOrigin.Width < targets[i].Width)
422 {
423 if (origins[i].X1 < actualPageArea.X1)
424 {
425 origins[i] = new Rectangle(origins[i].X0, origins[i].Y0, origins[i].X1 + 0.5 / zoom, origins[i].Y1);
426 }
427 else
428 {
429 origins[i] = new Rectangle(origins[i].X0 - 0.5 / zoom, origins[i].Y0, origins[i].X1, origins[i].Y1);
430 }
431 }
432
433
434 if (roundedOrigin.Height > targets[i].Height)
435 {
436 if (origins[i].Y0 > actualPageArea.Y0)
437 {
438 origins[i] = new Rectangle(origins[i].X0, origins[i].Y0 + 0.5 / zoom, origins[i].X1, origins[i].Y1);
439 }
440 else
441 {
442 origins[i] = new Rectangle(origins[i].X0, origins[i].Y0, origins[i].X1, origins[i].Y1 - 0.5 / zoom);
443 }
444 }
445 else if (roundedOrigin.Height < targets[i].Height)
446 {
447 if (origins[i].X1 < actualPageArea.X1)
448 {
449 origins[i] = new Rectangle(origins[i].X0, origins[i].Y0, origins[i].X1, origins[i].Y1 + 0.5 / zoom);
450 }
451 else
452 {
453 origins[i] = new Rectangle(origins[i].X0, origins[i].Y0 - 0.5 / zoom, origins[i].X1, origins[i].Y1);
454 }
455 }
456
457 roundedOrigin = origins[i].Round(zoom);
458
459 if (roundedOrigin.Width == oldRoundedOrigin.Width && roundedOrigin.Height == oldRoundedOrigin.Height && (roundedOrigin.Width != targets[i].Width || roundedOrigin.Height != targets[i].Height))
460 {
461 countBlanks++;
462 }
463
464 if (countBlanks >= 100)
465 {
466 //It seems that we can't coerce the expected size and the actual size to be the same. Give up.
467 return;
468 }
469 }
470 }
471
472 //Start each rendering thread.
473 for (int i = 0; i < destinations.Length; i++)
474 {
475 double dzoom = zoom;
476 Rectangle origin = origins[i];
477 if (this.ImageXRes != 72 || this.ImageYRes != 72)
478 {
479 dzoom *= Math.Sqrt(this.ImageXRes * this.ImageYRes) / 72;
480 origin = new Rectangle(origin.X0 * 72 / this.ImageXRes, origin.Y0 * 72 / this.ImageYRes, origin.X1 * 72 / this.ImageXRes, origin.Y1 * 72 / this.ImageYRes);
481 }
482
483 RenderingThreads[i].Render(Contexts[i].NativeContext, DisplayList, origin, (float)dzoom, destinations[i], pixelFormat, this.PageBounds, ClipToPageBounds);
484 }
485
486 //Wait until all the rendering threads have finished.
487 for (int i = 0; i < destinations.Length; i++)
488 {
489 RenderingThreads[i].WaitForRendering();
490 }
491 }

500 /// <summary>
501 /// Render the specified region to an image of the specified size, split in a number of tiles equal to the number of threads used by this <see cref="MuPDFMultiThreadedPageRenderer"/>, without marshaling. This method will not return until all the rendering threads have finished.
502 /// Since creating an array of <see cref="Span{T}"/> is not allowed, this method returns a delegate that accepts an integer parameter (representing the index of the span in the "array") and returns the <see cref="Span{T}"/> corresponding to that index.
503 /// </summary>
504 /// <param name="targetSize">The total size of the image that should be rendered.</param>
505 /// <param name="region">The region in page units that should be rendered.</param>
506 /// <param name="disposables">A collection of <see cref="IDisposable"/>s that can be used to free the memory where the rendered tiles are stored. You should keep track of these and dispose them when you have finished working with the images.</param>
507 /// <param name="pixelFormat">The format of the pixel data.</param>
508 /// <returns>A delegate that accepts an integer parameter (representing the index of the span in the "array") and returns the <see cref="Span{T}"/> corresponding to that index.</returns>
509 public GetSpanItem Render(RoundedSize targetSize, Rectangle region, out IDisposable[] disposables, PixelFormats pixelFormat)
510 {
511 RoundedRectangle[] targets = targetSize.Split(this.ThreadCount);
512
513 IntPtr[] destinations = new IntPtr[targets.Length];
514 disposables = new IDisposable[targets.Length];
515
516 bool hasAlpha = pixelFormat == PixelFormats.RGBA || pixelFormat == PixelFormats.BGRA;
517
518 int[] sizes = new int[destinations.Length];
519
520 for (int i = 0; i < targets.Length; i++)
521 {
522 int allocSize = targets[i].Width * targets[i].Height * (hasAlpha ? 4 : 3);
523
524 sizes[i] = allocSize;
525 destinations[i] = Marshal.AllocHGlobal(allocSize);
526 disposables[i] = new DisposableIntPtr(destinations[i], allocSize);
527 }
528
529 this.Render(targetSize, region, destinations, pixelFormat);
530
531 return i =>
532 {
533 unsafe
534 {
535 return new Span<byte>((void*)destinations[i], sizes[i]);
536 }
537 };
538 }
