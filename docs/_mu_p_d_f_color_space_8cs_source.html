24 /// <summary>
25 /// Colour space used by the image.
26 /// </summary>
27 public enum ColorSpaceType
28 {
29 /// <summary>
30 /// Not specified.
31 /// </summary>
32 None = 0,
33
34 /// <summary>
35 /// Grayscale. Each pixel is stored as a single byte.
36 /// </summary>
37 Gray = 1,
38
39 /// <summary>
40 /// RGB colour space. Each pixel is stored as three bytes representing the R, G, and B components of the colour in this order.
41 /// </summary>
42 RGB = 2,
43
44 /// <summary>
45 /// BGR colour space. Each pixel is stored as three bytes representing the B, G, and R components of the colour in this order.
46 /// </summary>
47 BGR = 3,
48
49 /// <summary>
50 /// CMYK colour space. Each pixel is stored as four bytes representing the C, M, Y, and K components of the colour in this order.
51 /// </summary>
52 CMYK = 4,
53
54 /// <summary>
55 /// Lab colour space. Each pixel is stored as three bytes representing the L*, a, and b components of the colour in this order.
56 /// </summary>
57 Lab = 5,
58
59 /// <summary>
60 /// Indexed colour space. Each pixel is stored as a single byte mapping to an entry in the colour space palette.
61 /// </summary>
62 Indexed = 6,
63
64 /// <summary>
65 /// Separation colour space.
66 /// </summary>
67 Separation = 7
68 }

117 internal static unsafe MuPDFColorSpace Create(IntPtr nativeContext, IntPtr nativePointer)
118 {
119 int csType = 0;
120 IntPtr baseColorSpace = IntPtr.Zero;
121 IntPtr lookupPointer = IntPtr.Zero;
122 int lookupLength = 0;
123 int nameLength = 0;
124
125 ExitCodes result = (ExitCodes)NativeMethods.GetColorSpaceData(nativeContext, nativePointer, ref csType, ref nameLength, ref baseColorSpace, ref lookupLength, ref lookupPointer);
126
127 switch (result)
128 {
129 case ExitCodes.EXIT_SUCCESS:
130 break;
131 case ExitCodes.ERR_COLORSPACE_METADATA:
132 throw new MuPDFException("An error occurred while retrieving colour space information!", result);
133 default:
134 throw new MuPDFException("Unknown error!", result);
135 }
136
137 byte[] nameArray = new byte[nameLength];
138
139 fixed (byte* nameArrayPtr = nameArray)
140 {
141 result = (ExitCodes)NativeMethods.GetColorSpaceName(nativeContext, nativePointer, nameLength, (IntPtr)nameArrayPtr);
142 }
143
144 switch (result)
145 {
146 case ExitCodes.EXIT_SUCCESS:
147 break;
148 case ExitCodes.ERR_COLORSPACE_METADATA:
149 throw new MuPDFException("An error occurred while retrieving colour space information!", result);
150 default:
151 throw new MuPDFException("Unknown error!", result);
152 }
153
154 string name = Encoding.UTF8.GetString(nameArray);
155
156 ColorSpaceType type = (ColorSpaceType)csType;
157
158 switch (type)
159 {
160 case ColorSpaceType.None:
161 return null;
162 case ColorSpaceType.Gray:
163 return new GrayscaleMuPDFColorSpace(name);
164 case ColorSpaceType.RGB:
165 return new RGBMuPDFColorSpace(name);
166 case ColorSpaceType.BGR:
167 return new BGRMuPDFColorSpace(name);
168 case ColorSpaceType.CMYK:
169 return new CMYKMuPDFColorSpace(name);
170 case ColorSpaceType.Lab:
171 return new LabMuPDFColorSpace(name);
172 case ColorSpaceType.Indexed:
173 {
174 MuPDFColorSpace baseMuPDFColorSpace = Create(nativeContext, baseColorSpace);
175 byte[] lookupTable = new byte[lookupLength * baseMuPDFColorSpace.NumBytes];
176
177 fixed (byte* lookupTablePtr = lookupTable)
178 {
179 Buffer.MemoryCopy((byte*)lookupPointer, lookupTablePtr, lookupLength * baseMuPDFColorSpace.NumBytes, lookupLength * baseMuPDFColorSpace.NumBytes);
180 }
181
182 return new IndexedMuPDFColorSpace(lookupTable, baseMuPDFColorSpace, name);
183 }
184 case ColorSpaceType.Separation:
185 {
186 string[] colorantNames = new string[lookupLength];
187
188 for (int i = 0; i < lookupLength; i++)
189 {
190 int colorantNameLength = 0;
191
192 result = (ExitCodes)NativeMethods.GetColorantNameLength(nativeContext, nativePointer, i, ref colorantNameLength);
193 switch (result)
194 {
195 case ExitCodes.EXIT_SUCCESS:
196 break;
197 case ExitCodes.ERR_COLORSPACE_METADATA:
198 throw new MuPDFException("An error occurred while retrieving colorant names!", result);
199 default:
200 throw new MuPDFException("Unknown error!", result);
201 }
202
203 byte[] colorantNameArray = new byte[colorantNameLength];
204
205 fixed (byte* colorantNamePtr = colorantNameArray)
206 {
207 result = (ExitCodes)NativeMethods.GetColorantName(nativeContext, nativePointer, i, colorantNameLength, (IntPtr)colorantNamePtr);
208 }
209
210 switch (result)
211 {
212 case ExitCodes.EXIT_SUCCESS:
213 break;
214 case ExitCodes.ERR_COLORSPACE_METADATA:
215 throw new MuPDFException("An error occurred while retrieving colorant names!", result);
216 default:
217 throw new MuPDFException("Unknown error!", result);
218 }
219
220 colorantNames[i] = Encoding.UTF8.GetString(colorantNameArray);
221 }
222
223 MuPDFColorSpace alternateColorSpace = MuPDFColorSpace.Create(nativeContext, baseColorSpace);
224
225 return new SeparationColorSpace(name, colorantNames, alternateColorSpace);
226 }
227 default:
228 throw new MuPDFException("Unknown colour space type!", ExitCodes.UNKNOWN_ERROR);
229 }
230 }
